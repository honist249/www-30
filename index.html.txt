function loadAllData() {
    document.querySelectorAll('input[type="checkbox"]').forEach(cb => {
        if (localStorage.getItem(cb.id) === 'true') cb.checked = true;
    });
    document.querySelectorAll('input[type="text"]').forEach(input => {
        input.value = localStorage.getItem(input.id) || '';
    });
    document.querySelectorAll('.editable-text').forEach(el => {
        const savedText = localStorage.getItem('text-edit-' + el.id);
        if (savedText) el.innerHTML = savedText;
    });
    updateAllProgress();
}

function saveData(key, value) {
    localStorage.setItem(key, value);
}

function showTab(tabId) {
    document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active'));
    document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));
    document.getElementById(tabId).classList.add('active');
    event.currentTarget.classList.add('active');
}

function updateAllProgress() {

        const checkboxes = document.querySelectorAll(selector);
        const completed = Array.from(checkboxes).filter(cb => cb.checked).length;
        const total = checkboxes.length;
        return total > 0 ? Math.round((completed / total) * 100) : 0;

        const bar = document.getElementById(barId);
        if (bar) {

            bar.textContent = percent + '%';
        }
    };
    
    personDetails.forEach(person => {
        updateBar(`p${person.id}_monthly_progress`, calculateProgress(person.id));
    });
    
    updateBar('monthlyOverallProgressBar', calculateProgress(null));
}

function resetAllProgress() {
    if (confirm("‡¶Ü‡¶™‡¶®‡¶ø ‡¶ï‡¶ø ‡¶∂‡ßÅ‡¶ß‡ßÅ‡¶Æ‡¶æ‡¶§‡ßç‡¶∞ ‡¶ï‡¶æ‡¶ú‡ßá‡¶∞ ‡¶Ö‡¶ó‡ßç‡¶∞‡¶ó‡¶§‡¶ø ‡¶è‡¶¨‡¶Ç ‡¶Æ‡¶®‡ßç‡¶§‡¶¨‡ßç‡¶Ø ‡¶Æ‡ßÅ‡¶õ‡ßá ‡¶´‡ßá‡¶≤‡¶§‡ßá ‡¶®‡¶ø‡¶∂‡ßç‡¶ö‡¶ø‡¶§? ‡¶Ü‡¶™‡¶®‡¶æ‡¶∞ ‡¶ï‡¶æ‡¶∏‡ßç‡¶ü‡¶Æ ‡¶≤‡ßá‡¶ñ‡¶æ‡¶ó‡ßÅ‡¶≤‡ßã ‡¶•‡ßá‡¶ï‡ßá ‡¶Ø‡¶æ‡¶¨‡ßá‡•§")) {
        const keysToDelete = [];
        // First, gather all the keys we want to remove.
        for (let i = 0; i < localStorage.length; i++) {
            const key = localStorage.key(i);
            if (key.startsWith('task-') || key.startsWith('comment-')) {
                keysToDelete.push(key);
            }
        }

        // Now, remove them. This is safer than modifying while iterating.
        for (const key of keysToDelete) {
            localStorage.removeItem(key);
        }

        alert("‡¶ï‡¶æ‡¶ú‡ßá‡¶∞ ‡¶Ö‡¶ó‡ßç‡¶∞‡¶ó‡¶§‡¶ø ‡¶è‡¶¨‡¶Ç ‡¶Æ‡¶®‡ßç‡¶§‡¶¨‡ßç‡¶Ø ‡¶∞‡¶ø‡¶∏‡ßá‡¶ü ‡¶ï‡¶∞‡¶æ ‡¶π‡¶Ø‡¶º‡ßá‡¶õ‡ßá‡•§ ‡¶™‡ßÉ‡¶∑‡ßç‡¶†‡¶æ‡¶ü‡¶ø ‡¶™‡ßÅ‡¶®‡¶∞‡¶æ‡¶Ø‡¶º ‡¶≤‡ßã‡¶° ‡¶π‡¶¨‡ßá‡•§");
        location.reload();
    }
}
